async function compare() { const pickup = document.getElementById("pickup").value; const drop = document.getElementById("drop").value; if (!pickup || !drop) { alert("Please enter both locations"); return; } document.getElementById("results").innerHTML = "Calculating..."; try { const res = await fetch(`/api/distance?pickup=${pickup}&drop=${drop}`); const data = await res.json(); const distanceText = data.rows[0].elements[0].distance.text; const distance = parseFloat(distanceText); const uber = 50 + (12 * distance); const ola = 40 + (11 * distance); const rapido = 30 + (8 * distance); document.getElementById("results").innerHTML = `
Uber: ₹${uber.toFixed(0)}
Ola: ₹${ola.toFixed(0)}
Rapido: ₹${rapido.toFixed(0)}
`; } catch (err) { document.getElementById("results").innerHTML = "Error fetching data"; } }
